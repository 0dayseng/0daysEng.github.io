16 static inline void kvm_skip_instr(struct kvm_vcpu *vcpu)
17 {
18  if (vcpu_mode_is_32bit(vcpu)) {
19  kvm_skip_instr32(vcpu);
20  } else {
21  *vcpu_pc(vcpu) += 4;
22  *vcpu_cpsr(vcpu) &= ~PSR_BTYPE_MASK;
23  }

25  /* advance the singlestep state machine */
